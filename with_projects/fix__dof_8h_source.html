82  RHSVECTOR &b) {
83  const lf::assemble::size_type N(A.cols());
84  LF_ASSERT_MSG(A.rows() == N, "Matrix must be square!");
85  LF_ASSERT_MSG(N == b.size(),
86  "Mismatch N = " << N << " <-> b.size() = " << b.size());
87  {
88  // Multiply sparse matrix with the vector of fixed components and subtract
89  // result from right hand side
90  // To skip irrelevant components of fixed_vec rely on a temporary vector
91  Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> tmp_vec(N);
92  for (lf::assemble::gdof_idx_t k = 0; k < N; ++k) {
93  const auto selval{selectvals(k)};
94  if (selval.first) {
95  tmp_vec[k] = selval.second;
96  } else {
97  tmp_vec[k] = SCALAR();
98  }
99  }
100  A.MatVecMult(-1.0, tmp_vec, b);
101  }
102  // Set vector components of right-hand-side vector for prescribed values
103  for (lf::assemble::gdof_idx_t k = 0; k < N; ++k) {
104  const auto selval{selectvals(k)};
105  if (selval.first) {
106  b[k] = selval.second;
107  }
108  }
109  // Set rows and columns of the sparse matrix corresponding to the fixed
110  // solution components to zero
111  A.setZero([&selectvals](gdof_idx_t i, gdof_idx_t j) {
112  return (selectvals(i).first || selectvals(j).first);
113  });
114  // Old implementation, demonstrating what is going on
115  // lf::assemble::COOMatrix<double>::TripletVec::iterator new_last =
116  // std::remove_if(
117  // A.triplets().begin(), A.triplets().end(),
118  // [&fixed_comp_flags](
119  // typename lf::assemble::COOMatrix<double>::Triplet &triplet) {
120  // return (fixed_comp_flags[triplet.row()] ||
121  // fixed_comp_flags[triplet.col()]);
122  // });
123  // // Adjust size of triplet vector
124  // A.triplets().erase(new_last, A.triplets().end());
125  // Add Unit diagonal entries corrresponding to fixed components
126  for (lf::assemble::gdof_idx_t dofnum = 0; dofnum < N; ++dofnum) {
127  const auto selval{selectvals(dofnum)};
128  if (selval.first) {
129  A.AddToEntry(dofnum, dofnum, 1.0);
130  }
