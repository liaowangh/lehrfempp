132 template <typename DOFVECTOR>
133 double MeshFunctionL2NormDifference<FUNCTOR>::operator()(
134  const lf::mesh::Entity &cell, const DOFVECTOR &dofs) {
135  using dof_t = typename DOFVECTOR::value_type;
136  // Topological type of the cell
137  const lf::base::RefEl ref_el{cell.RefEl()};
138  auto &fe = fe_precomp_[ref_el.Id()];
139  LF_ASSERT_MSG(fe.isInitialized(),
140  "No Shape functions available for " << ref_el);
141  // Query the shape of the cell
142  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
143  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
144  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
145  "Only 2D implementation available!");
146  LF_ASSERT_MSG(dofs.size() >= fe.NumRefShapeFunctions(),
147  "Dof vector too short");
148  // Obtain cell--type depend invariant data
149  const Eigen::MatrixXd &qr_Points{fe.Qr().Points()};
150  const Eigen::VectorXd &qr_Weights{fe.Qr().Weights()};
151  const size_type qr_NumPts = fe.Qr().NumPoints();
152  const size_type num_LSF = fe.NumRefShapeFunctions();
153  const auto &rsf_QuadPts{fe.PrecompReferenceShapeFunctions()};
154 
155  SWITCHEDSTATEMENT(ctrl_, kout_cell,
156  std::cout << ref_el << ", (Nlsf = " << num_LSF
157  << ", Nqr = " << qr_NumPts << ") shape = \n"
158  << geo_ptr->Global(ref_el.NodeCoords())
159  << std::endl);
160 
161  // World coordinates of quadrature points
162  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_Points));
163  // Obtain the metric factors for the quadrature points
164  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_Points));
165  SWITCHEDSTATEMENT(
166  ctrl_, kout_comp, std::cout << "\t dofs = "; for (auto x
167  : dofs) {
168  std::cout << x << " ";
169  } std::cout << "\t dets "
170  << determinants.transpose() << std::endl);
171 
172  double sum = 0.0; // summation variable
173 
174  auto uvals = u_(cell, qr_Points);
175 
176  // Loop over quadrature points
177  for (int k = 0; k < qr_NumPts; ++k) {
178  // Value of finite element function at quadrature point
179  // Loop over local shape functions/dofs to compute the value of
180  // the finite element function in the quadrature point
181  dof_t uh_val = 0.0;
182  for (int j = 0; j < num_LSF; ++j) {
183  uh_val += rsf_QuadPts(j, k) * dofs[j];
184  }
185  // form the difference
186  uh_val -= uvals[k];
187  // sum the quared modulus weighted with quadrature weight and metric factor
188  sum += qr_Weights[k] * determinants[k] * std::fabs(uh_val * uh_val);
189  SWITCHEDSTATEMENT(ctrl_, kout_comp,
190  std::cout << "\t @ " << (mapped_qpts.col(k)).transpose()
191  << ": uh = " << uh_val << ", u = " << uvals[k]
192  << std::endl);
193  }
194  SWITCHEDSTATEMENT(ctrl_, kout_comp,
195  std::cout << "\t loc norm^2 = " << sum << std::endl);
196  return sum;
197 }

314 double MeshFunctionL2GradientDifference<VEC_FUNC>::operator()(
315  const lf::mesh::Entity &cell, const DOFVECTOR &dofs) {
316  using dof_t = typename DOFVECTOR::value_type;
317  // Topological type of the cell
318  const lf::base::RefEl ref_el{cell.RefEl()};
319  auto &pfe = fe_precomp_[ref_el.Id()];
320  LF_ASSERT_MSG(pfe.isInitialized(), "No ShapeFunctions for " << ref_el);
321  // Query the shape of the cell
322  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
323  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
324  LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
325  "Only 2D implementation available!");
326  LF_ASSERT_MSG(dofs.size() >= pfe.NumRefShapeFunctions(),
327  "Dof vector too short");
328 
329  // Physical dimension of the cell
330  const dim_t world_dim = geo_ptr->DimGlobal();
331 
332  // Obtain cell--type depend but cell-independent data
333  auto &qr_Points{pfe.Qr().Points()};
334  auto &qr_Weights{pfe.Qr().Weights()};
335  size_type qr_NumPts = pfe.Qr().NumPoints();
336  size_type num_LSF = pfe.NumRefShapeFunctions();
337  auto gradrsf_QuadPts = pfe.PrecompGradientsReferenceShapeFunctions();
338 
339  SWITCHEDSTATEMENT(ctrl_, kout_cell,
340  std::cout << ref_el << ", (Nlsf = " << num_LSF
341  << ", Nqr = " << qr_NumPts << ") shape = \n"
342  << geo_ptr->Global(ref_el.NodeCoords())
343  << std::endl);
344 
345  // World coordinates of quadrature points
346  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_Points));
347  // Obtain the metric factors for the quadrature points
348  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_Points));
349  // Fetch the transformation matrices for gradients
350  const Eigen::MatrixXd JinvT(geo_ptr->JacobianInverseGramian(qr_Points));
351 
352  SWITCHEDSTATEMENT(
353  ctrl_, kout_comp, std::cout << "\t dofs = "; for (auto x
354  : dofs) {
355  std::cout << x << " ";
356  } std::cout << "\t dets "
357  << determinants.transpose() << std::endl);
358 
359  double sum = 0.0; // summation variable
360 
361  auto vfval = vecfield_(cell, qr_Points);
362 
363  // Loop over quadrature points
364  for (unsigned k = 0; k < qr_NumPts; ++k) {
365  // Value of the gradient of the finite element function at quadrature point
366  // Transformed gradients
367  const auto trf_grad(
368  JinvT.block(0, 2 * k, world_dim, 2) *
369  gradrsf_QuadPts.block(0, 2 * k, gradrsf_QuadPts.rows(), 2).transpose());
370  // Loop over local shape functions/dofs to compute the value of
371  // the finite element function in the quadrature point
372  Eigen::Matrix<dof_t, Eigen::Dynamic, 1> grad_uh_val(world_dim, 1);
373  grad_uh_val.setZero();
374  for (unsigned j = 0; j < num_LSF; ++j) {
375  grad_uh_val += (dofs[j] * trf_grad.col(j));
376  }
377  // form the difference
378  for (int l = 0; l < world_dim; ++l) {
379  grad_uh_val[l] -= vfval[k][l];
380  }
381  // sum the quared modulus weighted with quadrature weight and metric factor
382  sum += qr_Weights[k] * determinants[k] * grad_uh_val.squaredNorm();
383  // clang-format on
384  SWITCHEDSTATEMENT(ctrl_, kout_comp,
385  std::cout << "\t @ " << (mapped_qpts.col(k)).transpose()
386  << ": grad uh = [" << grad_uh_val.transpose()
387  << "[, vf = [" << vfval[k][0] << ' '
388  << vfval[k][1] << "]" << std::endl);
389  // clang-format on
390  }
391  SWITCHEDSTATEMENT(ctrl_, kout_comp,
392  std::cout << "\t loc norm^2 = " << sum << std::endl);
393  return sum;
394 }
