409  public:
433  template <typename LOCALDOFINFO>
434  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
435  LOCALDOFINFO &&locdof)
436  : mesh_p_(std::move(mesh_p)) {
437  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
438 
439  // Index counter for global shape functions = global dof
440  gdof_idx_t dof_idx = 0;
441 
442  // Step I: Set indices for shape functions on nodes
443  // Run through node indices (entities of co-dimension 2)
444  const size_type no_nodes = mesh_p_->NumEntities(2);
445  num_int_dofs_[2].resize(no_nodes, 0);
446  offsets_[2].resize(no_nodes + 1, 0);
447  // Traverse nodes (co-dimension-2 entities) based on indices
448  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
449  // Obtain pointer to node entity
450  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
451  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
452  // Offset for indices of node in index vector
453  glb_idx_t node_dof_offset = dof_idx;
454  offsets_[2][node_idx] = node_dof_offset;
455  // Request number of local shape functions associated with the node
456  size_type no_int_dof_node = locdof(*node_p);
457  num_int_dofs_[2][node_idx] = no_int_dof_node;
458 
459  // Store dof indices in array
460  for (unsigned j = 0; j < no_int_dof_node; j++) {
461  dofs_[2].push_back(dof_idx);
462  dof_entities_.push_back(node_p); // Store entity for current dof
463  dof_idx++; // Move on to next index
464  }
465  }
466  // Set sentinel
467  offsets_[2][no_nodes] = dof_idx;
468 
469  // Step II: Set indices for shape functions on edges (co-dimension = 1)
470  const size_type no_edges = mesh_p_->NumEntities(1);
471  // Set length of edge-related index vectors
472  num_int_dofs_[1].resize(no_edges, 0);
473  offsets_[1].resize(no_edges + 1, 0);
474  // points to the position of the current dof index
475  size_type edge_dof_offset = 0;
476  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
477  // Obtain pointer to edge entity
478  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
479  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
480  // Offset for indices of edge dof in index vector
481  offsets_[1][edge_idx] = edge_dof_offset;
482  size_type no_int_dof_edge = locdof(*edge);
483  num_int_dofs_[1][edge_idx] = no_int_dof_edge;
484 
485  // Obtain indices for basis functions sitting at endpoints
486  // Endpoints are mesh entities with co-dimension = 2
487  for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
488  const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
489  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
490  size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
491  // Copy indices of shape functions from nodes to edge
492  for (unsigned j = 0; j < no_int_dofs_ep; j++) {
493  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
494  edge_dof_offset++;
495  }
496  }
497  // Set indices for interior edge degrees of freedom
498  for (unsigned j = 0; j < no_int_dof_edge; j++) {
499  dofs_[1].push_back(dof_idx);
500  edge_dof_offset++;
501  dof_entities_.push_back(edge);
502  dof_idx++;
503  }
504  }
505  // Set sentinel
506  offsets_[1][no_edges] = edge_dof_offset;
507 
508  // Step III: Set indices for shape functions on cells (co-dimension = 0)
509  const size_type no_cells = mesh_p_->NumEntities(0);
510  // Set length of cell-related index vectors
511  num_int_dofs_[0].resize(no_cells, 0);
512  offsets_[0].resize(no_cells + 1, 0);
513  // points to the position of the current dof index
514  size_type cell_dof_offset = 0;
515  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
516  // Obtain pointer to current ell
517  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
518  // Offset for indices of cell dof in index vector
519  offsets_[0][cell_idx] = cell_dof_offset;
520  size_type no_int_dof_cell = locdof(*cell);
521  num_int_dofs_[0][cell_idx] = no_int_dof_cell;
522 
523  // Obtain indices for basis functions in vertices
524  for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
525  const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
526  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
527  size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
528  // Copy indices of shape functions from nodes to cell
529  for (unsigned j = 0; j < no_int_dofs_vt; j++) {
530  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
531  cell_dof_offset++;
532  }
533  }
534 
535  // Collect indices of interior shape functions of edges
536  // Internal ordering will depend on the orientation of the edge
537  auto edge_orientations = cell->RelativeOrientations();
538  auto edges = cell->SubEntities(1);
539  // Loop over edges
540  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
541  ed_sub_idx++) {
542  const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
543  const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
544  const glb_idx_t edge_int_dof_offset =
545  offsets_[1][edge_idx + 1] - no_int_dof_edge;
546 
547  // Copy indices of shape functions from edges to cell
548  // The order, in which they are copied depends on the relative
549  // orientation of the edge w.r.t. the cell
550  switch (edge_orientations[ed_sub_idx]) {
551  case lf::mesh::Orientation::positive: {
552  for (int j = 0; j < no_int_dof_edge; j++) {
553  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
554  cell_dof_offset++;
555  }
556  break;
557  }
558  case lf::mesh::Orientation::negative: {
559  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
560  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
561  cell_dof_offset++;
562  }
563  break;
564  }
565  } // end switch
566  } // end loop over edges
567 
568  // Set indices for interior shape functions of the cell
569  for (unsigned j = 0; j < no_int_dof_cell; j++) {
570  dofs_[0].push_back(dof_idx);
571  cell_dof_offset++;
572  dof_entities_.push_back(cell);
573  dof_idx++;
574  } // end loop over interior dof of cell
575  } // end loop over cells
576  // Set sentinel
577  offsets_[0][no_cells] = cell_dof_offset;
578 
579  // Finally set number of global shape functions
580  num_dof_ = dof_idx;
581  } // end constructor
