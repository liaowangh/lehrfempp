254 template <typename SCALAR, typename DIFF_COEFF, typename REACTION_COEFF>
255 typename lf::uscalfe::ReactionDiffusionElementMatrixProvider<
256  SCALAR, DIFF_COEFF, REACTION_COEFF>::ElemMat
257 ReactionDiffusionElementMatrixProvider<
258  SCALAR, DIFF_COEFF, REACTION_COEFF>::Eval(const lf::mesh::Entity &cell) {
259  // Topological type of the cell
260  const lf::base::RefEl ref_el{cell.RefEl()};
261  // Obtain precomputed information about values of local shape functions
262  // and their gradients at quadrature points.
263  PrecomputedScalarReferenceFiniteElement<SCALAR> &pfe =
264  fe_precomp_[ref_el.Id()];
265  // Accident: cell is of a type not coverence by finite element specifications
266  LF_ASSERT_MSG(
267  pfe.isInitialized(),
268  "No local shape function information for entity type " << ref_el);
269 
270  // Query the shape of the cell
271  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
272  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
273  LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
274  "Only 2D implementation available!");
275  SWITCHEDSTATEMENT(ctrl_, kout_cell,
276  std::cout << ref_el << ", shape = \n"
277  << geo_ptr->Global(ref_el.NodeCoords())
278  << std::endl);
279  // Physical dimension of the cell
280  const dim_t world_dim = geo_ptr->DimGlobal();
281 
282  const Eigen::VectorXd determinants(
283  geo_ptr->IntegrationElement(pfe.Qr().Points()));
284  LF_ASSERT_MSG(
285  determinants.size() == pfe.Qr().NumPoints(),
286  "Mismatch " << determinants.size() << " <-> " << pfe.Qr().NumPoints());
287  // Fetch the transformation matrices for the gradients
288  const Eigen::MatrixXd JinvT(
289  geo_ptr->JacobianInverseGramian(pfe.Qr().Points()));
290  LF_ASSERT_MSG(
291  JinvT.cols() == 2 * pfe.Qr().NumPoints(),
292  "Mismatch " << JinvT.cols() << " <-> " << 2 * pfe.Qr().NumPoints());
293  LF_ASSERT_MSG(JinvT.rows() == world_dim,
294  "Mismatch " << JinvT.rows() << " <-> " << world_dim);
295 
296  // compute values of alpha, gamma at quadrature points:
297  auto alphaval = alpha_(cell, pfe.Qr().Points());
298  auto gammaval = gamma_(cell, pfe.Qr().Points());
299 
300  // Element matrix
301  ElemMat mat(pfe.NumRefShapeFunctions(), pfe.NumRefShapeFunctions());
302  mat.setZero();
303 
304  // Loop over quadrature points
305  for (base::size_type k = 0; k < pfe.Qr().NumPoints(); ++k) {
306  const double w = pfe.Qr().Weights()[k] * determinants[k];
307  // Transformed gradients
308  const auto trf_grad(JinvT.block(0, 2 * k, world_dim, 2) *
309  pfe.PrecompGradientsReferenceShapeFunctions()
310  .block(0, 2 * k, mat.rows(), 2)
311  .transpose());
312  // Transformed gradients multiplied with coefficient
313  const auto alpha_trf_grad(alphaval[k] * trf_grad);
314  mat += w * (alpha_trf_grad.transpose() * trf_grad +
315  (gammaval[k] * pfe.PrecompReferenceShapeFunctions().col(k)) *
316  (pfe.PrecompReferenceShapeFunctions().col(k).transpose()));
317  }
318  return mat;
319 }

463  const lf::base::RefEl ref_el{edge.RefEl()};
464  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
465  "Edge must be of segment type");
466  // Query the shape of the edge
467  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
468  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
469 
470  // Obtain the metric factors for the quadrature points
471  const Eigen::VectorXd determinants(
472  geo_ptr->IntegrationElement(fe_precomp_.Qr().Points()));
473  LF_ASSERT_MSG(determinants.size() == fe_precomp_.Qr().NumPoints(),
474  "Mismatch " << determinants.size() << " <-> "

476 
477  // Element matrix
478  ElemMat mat(fe_precomp_.NumRefShapeFunctions(),
479  fe_precomp_.NumRefShapeFunctions());
480  mat.setZero();
481 
482  auto gammaval = gamma_(edge, fe_precomp_.Qr().Points());
483 
484  // Loop over quadrature points
485  for (long k = 0; k < determinants.size(); ++k) {
486  // Build local matrix by summing rank-1 contributions
487  // from quadrature points.
488  const auto w =
489  (fe_precomp_.Qr().Weights()[k] * determinants[k]) * gammaval[k];
490  mat += ((fe_precomp_.PrecompReferenceShapeFunctions().col(k)) *
491  (fe_precomp_.PrecompReferenceShapeFunctions().col(k).transpose())) *
492  w;
493  }

847 typename ScalarLoadEdgeVectorProvider<SCALAR, FUNCTOR, EDGESELECTOR>::ElemVec
848 ScalarLoadEdgeVectorProvider<SCALAR, FUNCTOR, EDGESELECTOR>::Eval(
849  const lf::mesh::Entity &edge) {
850  // Topological type of the cell
851  const lf::base::RefEl ref_el{edge.RefEl()};
852  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
853  "Edge must be of segment type");
854  // Query the shape of the edge
855  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
856  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
857 
858  // Quadrature points on physical edge
859  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(pfe_.Qr().Points()));
860  LF_ASSERT_MSG(
861  mapped_qpts.cols() == pfe_.Qr().NumPoints(),
862  "Mismatch " << mapped_qpts.cols() << " <-> " << pfe_.Qr().NumPoints());
863 
864  // Obtain the metric factors for the quadrature points
865  const Eigen::VectorXd determinants(
866  geo_ptr->IntegrationElement(pfe_.Qr().Points()));
867  LF_ASSERT_MSG(
868  determinants.size() == pfe_.Qr().NumPoints(),
869  "Mismatch " << determinants.size() << " <-> " << pfe_.Qr().NumPoints());
870 
871  // Element vector
872  ElemVec vec(pfe_.NumRefShapeFunctions());
873  vec.setZero();
874 
875  auto g_vals = g_(edge, pfe_.Qr().Points());
876 
877  // Loop over quadrature points
878  for (base::size_type k = 0; k < pfe_.Qr().NumPoints(); ++k) {
879  // Add contribution of quadrature point to local vector
880  const auto w = (pfe_.Qr().Weights()[k] * determinants[k]) * g_vals[k];
881  vec += pfe_.PrecompReferenceShapeFunctions().col(k) * w;
882  }
883  return vec;
884 }
