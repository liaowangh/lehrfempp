166 typename LinearFELocalLoadVector<SCALAR, FUNCTOR>::ElemVec
167 LinearFELocalLoadVector<SCALAR, FUNCTOR>::Eval(
168  const lf::mesh::Entity &cell) const {
169  // Topological type of the cell
170  const lf::base::RefEl ref_el{cell.RefEl()};
171  const lf::base::size_type num_nodes{ref_el.NumNodes()};
172 
173  // Obtain the vertex coordinates of the cell, which completely
174  // describe its shape.
175  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
176  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
177  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
178  "Only 2D implementation available!");
179  const Eigen::MatrixXd &ref_el_corners(ref_el.NodeCoords());
180  // World coordinates of vertices
181  // const Eigen::MatrixXd vertices{geo_ptr->Global(ref_el_corners)};
182  // Debugging output
183  SWITCHEDSTATEMENT(dbg_ctrl, dbg_geo,
184  std::cout << ref_el << ", shape = \n"
185  << (geo_ptr->Global(ref_el_corners))
186  << std::endl);
187  // Midpoints of edges in the reference cell
188  Eigen::MatrixXd ref_mp(2, 4);
189  switch (ref_el) {
190  case lf::base::RefEl::kTria(): {
191  // clang-format off
192  ref_mp << 0.5,0.5,0.0,-1.0,
193  0.0,0.5,0.5,-1.0;
194  // clang-format on
195  break;
196  }
197  case lf::base::RefEl::kQuad(): {
198  // clang-format off
199  ref_mp << 0.5,1,0.5,0.0,
200  0.0,0.5,1.0,0.5;
201  // clang-format on
202  break;
203  }
204  default: {
205  LF_ASSERT_MSG(false, "Illegal entity type!");
206  break;
207  }
208  } // end switch
209 
210  const double area = lf::geometry::Volume(*geo_ptr);
211 
212  // Vector for returning element vector
213  ElemVec elem_vec = ElemVec::Zero();
214  // Run over the midpoints of edges and fetch values of the source function
215  // there
216  auto fvals = f_(cell, ref_mp);
217 
218  for (int k = 0; k < num_nodes; k++) {
219  elem_vec[k] += 0.5 * fvals[k];
220  elem_vec[(k + 1) % num_nodes] += 0.5 * fvals[k];
221  }
222  SWITCHEDSTATEMENT(
223  dbg_ctrl, dbg_locvec,
224  std::cout << "element vector = " << elem_vec.head(num_nodes).transpose()
225  << std::endl);
226  return (area / num_nodes) * elem_vec;
227 }
