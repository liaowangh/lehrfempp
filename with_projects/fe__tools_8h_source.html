34 template <class MF, class QR_SELECTOR>
35 auto LocalIntegral(const mesh::Entity &e, const QR_SELECTOR &qr_selector,
36  const MF &mf) -> MeshFunctionReturnType<MF> {
37  using MfType = MeshFunctionReturnType<MF>;
38  auto qr = qr_selector(e);
39  auto values = mf(e, qr.Points());
40  auto weights_ie =
41  (qr.Weights().cwiseProduct(e.Geometry()->IntegrationElement(qr.Points())))
42  .eval();
43  LF_ASSERT_MSG(values.size() == qr.NumPoints(),
44  "mf returns vector with wrong size.");
45  if constexpr (std::is_arithmetic_v<MfType>) {
46  auto value_m = Eigen::Map<Eigen::Matrix<MfType, 1, Eigen::Dynamic>>(
47  &values[0], 1, values.size());
48  return (value_m * weights_ie)(0);
49  }
50 
51  if constexpr (base::is_eigen_matrix<MfType>) { // NOLINT
52  constexpr int size = MfType::SizeAtCompileTime;
53  if constexpr (size != Eigen::Dynamic) {
54  auto value_m = Eigen::Map<
55  Eigen::Matrix<typename MfType::Scalar, size, Eigen::Dynamic>>(
56  &values[0](0, 0), size, values.size());
57  MfType result;
58  auto result_m =
59  Eigen::Map<Eigen::Matrix<typename MfType::Scalar, size, 1>>(
60  &result(0, 0));
61  result_m = value_m * weights_ie;
62  return result;
63  }
64  }
65  // fallback: we cannot make any optimizations:
66  MfType temp = weights_ie(0) * values[0];
67  for (Eigen::Index i = 1; i < qr.NumPoints(); ++i) {
68  temp = temp + weights_ie(i) * values[i];
69  }
70  return temp;
71 }
72 }; // namespace internal

106 template <class MF, class QR_SELECTOR,
107  class ENTITY_PREDICATE = base::PredicateTrue>
108 auto IntegrateMeshFunction(const lf::mesh::Mesh &mesh, const MF &mf,
109  const QR_SELECTOR &qr_selector,
110  const ENTITY_PREDICATE &ep = base::PredicateTrue{},
111  int codim = 0) -> MeshFunctionReturnType<MF> {
112  static_assert(isMeshFunction<MF>);
113  using MfType = MeshFunctionReturnType<MF>;
114 
115  auto entities = mesh.Entities(codim);
116  auto result = internal::LocalIntegral(**entities.begin(), qr_selector, mf);
117  for (auto i = entities.begin() + 1; i != entities.end(); ++i) {
118  if (!ep(**i)) {
119  continue;
120  }
121  result = result + internal::LocalIntegral(**i, qr_selector, mf);
122  }
123  return result;
124 }

201 template <typename SCALAR, typename MF, typename SELECTOR = base::PredicateTrue>
202 auto NodalProjection(const UniformScalarFESpace<SCALAR> &fe_space, MF &&u,
203  SELECTOR &&pred = base::PredicateTrue{}) {
204  static_assert(isMeshFunction<std::remove_reference_t<MF>>);
205  // choose scalar type so it can hold the scalar type of u as well as
206  // SCALAR
207  using scalarMF_t = MeshFunctionReturnType<std::remove_reference_t<MF>>;
208  using scalar_t = decltype(SCALAR(0) * scalarMF_t(0));
209  // Return type, type for FE coefficient vector
210  using dof_vec_t = Eigen::Matrix<scalar_t, Eigen::Dynamic, 1>;
211 
212  // Underlying mesh instance
213  const lf::mesh::Mesh &mesh{*fe_space.Mesh()};
214  // Fetch local-to-global index mapping for shape functions
215  const lf::assemble::DofHandler &dofh{fe_space.LocGlobMap()};
216 
217  // Vector for returning expansion coefficients
218  dof_vec_t glob_dofvec(dofh.NumDofs());
219  glob_dofvec.setZero();
220 
221  // Loop over all cells
222  for (const lf::mesh::Entity *cell : mesh.Entities(0)) {
223  if (!pred(*cell)) {
224  continue;
225  }
226  // Topological type of the cell
227  const lf::base::RefEl ref_el{cell->RefEl()};
228 
229  // TODO(ralfh) uncommend when ctrl_prj is well-defined.
230  // SWITCHEDSTATEMENT(ctrl_prj, kout_prj_cell,
231  // std::cout << ref_el << ", shape = \n"
232  // << geo_ptr->Global(ref_el.NodeCoords())
233  // << std::endl);
234 
235  // Information about local shape functions on reference element
236  auto ref_shape_fns = fe_space.ShapeFunctionLayout(ref_el);
237  LF_ASSERT_MSG(ref_shape_fns, "reference shape function for "
238  << ref_el << " not available.");
239  // Obtain reference coordinates for evaluation nodes
240  const Eigen::MatrixXd ref_nodes(ref_shape_fns->EvaluationNodes());
241 
242  // Collect values of function to be projected in a row vector
243  auto uvalvec = u(*cell, ref_nodes);
244 
245  // Compute the resulting local degrees of freedom
246  auto dofvec(ref_shape_fns->NodalValuesToDofs(
247  Eigen::Map<Eigen::Matrix<scalar_t, 1, Eigen::Dynamic>>(
248  &uvalvec[0], 1, uvalvec.size())));
249 
250  // Set the corresponing global degrees of freedom
251  // Note: "Setting", not "adding to"
252  // Number of local shape functions
253  const size_type num_loc_dofs = dofh.NumLocalDofs(*cell);
254  LF_ASSERT_MSG(
255  dofvec.size() == num_loc_dofs,
256  "Size mismatch: " << dofvec.size() << " <-> " << num_loc_dofs);
257  // Fetch global numbers of local shape functions
258  nonstd::span<const lf::assemble::gdof_idx_t> ldof_gidx(
259  dofh.GlobalDofIndices(*cell));
260  // Insert dof values into the global coefficient vector
261  for (int j = 0; j < num_loc_dofs; ++j) {
262  glob_dofvec[ldof_gidx[j]] = dofvec[j];
263  }
264  }
265  return glob_dofvec;
266 }

308 template <typename SCALAR, typename EDGESELECTOR, typename FUNCTION>
309 std::vector<std::pair<bool, SCALAR>> InitEssentialConditionFromFunction(
310  const lf::assemble::DofHandler &dofh,
311  const ScalarReferenceFiniteElement<SCALAR> &fe_spec_edge,
312  EDGESELECTOR &&esscondflag, FUNCTION &&g) {
313  static_assert(isMeshFunction<std::remove_reference_t<FUNCTION>>);
314  // static_assert(isMeshFunction<FUNCTION>, "g must by a MeshFunction object");
315  LF_ASSERT_MSG(fe_spec_edge.RefEl() == lf::base::RefEl::kSegment(),
316  "finite element specification must be for an edge!");
317 
318  // *** I: Preprocessing ***
319  // Fetch numbers of evaluation nodes and local shape functions
320  const size_type num_eval_pts = fe_spec_edge.NumEvaluationNodes();
321  const size_type num_rsf = fe_spec_edge.NumRefShapeFunctions();
322 
323  // Preprocessing: obtain evaluation nodes on reference segment [0,1]
324  const Eigen::MatrixXd ref_eval_pts{fe_spec_edge.EvaluationNodes()};
325  Eigen::MatrixXd eval_pts(2, num_eval_pts);
326  Eigen::Matrix<SCALAR, 1, Eigen::Dynamic> dof_vals(num_eval_pts);
327 
328  // Underlying mesh
329  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
330 
331  // Vector for returning flags and fixed values for dofs
332  const size_type num_coeffs = dofh.NumDofs();
333  std::vector<std::pair<bool, SCALAR>> flag_val_vec(num_coeffs,
334  {false, SCALAR{}});
335 
336  // *** II: Local computations ****
337  // Visit all edges of the mesh (codim-1 entities)
338  for (const lf::mesh::Entity *edge : mesh.Entities(1)) {
339  // Check whether the current edge carries dofs to be imposed by the
340  // function g. The decision relies on the predicate `esscondflag`
341  if (esscondflag(*edge)) {
342  // Evaluate mesh function at several points specified by their
343  // reference coordinates.
344  auto g_vals = g(*edge, ref_eval_pts);
345 
346  // Compute degrees of freedom from function values in evaluation
347  // points
348  dof_vals = fe_spec_edge.NodalValuesToDofs(
349  Eigen::Map<Eigen::Matrix<SCALAR, 1, Eigen::Dynamic>>(&g_vals[0], 1,
350  g_vals.size()));
351  LF_ASSERT_MSG(dof_vals.size() == num_rsf,
352  "Mismatch " << dof_vals.size() << " <-> " << num_rsf);
353  LF_ASSERT_MSG(
354  dofh.NumLocalDofs(*edge) == num_rsf,
355  "Mismatch " << dofh.NumLocalDofs(*edge) << " <-> " << num_rsf);
356  // Fetch indices of global shape functions associated with current
357  // edge
358  auto gdof_indices{dofh.GlobalDofIndices(*edge)};
359  int k = 0;
360  // Set flags and values; setting, no accumulation here!
361  for (const lf::assemble::gdof_idx_t gdof_idx : gdof_indices) {
362  flag_val_vec[gdof_idx] = {true, dof_vals[k++]};
363  }
364  }
365  }
366  return flag_val_vec;
367 }

398 template <typename LOC_COMP, typename COEFFVECTOR, typename SELECTOR>
399 double SumCellFEContrib(const lf::assemble::DofHandler &dofh,
400  LOC_COMP &loc_comp, const COEFFVECTOR &uh,
401  SELECTOR &&pred) {
402  // Type for passing local coefficient vectors
403  using dofvector_t = std::vector<typename COEFFVECTOR::value_type>;
404  // Retrieve the mesh
405  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
406 
407  // Check whether sufficiently large vector uh
408  LF_ASSERT_MSG(uh.size() >= dofh.NumDofs(), "uh vector too short!");
409 
410  double sum = 0.0;
412  for (const lf::mesh::Entity *cell : mesh.Entities(0)) {
413  if (pred(*cell)) {
414  // Number of local shape functions
415  const size_type num_loc_dofs = dofh.NumLocalDofs(*cell);
416  // Allocate a temporary vector of appropriate size
417  dofvector_t loc_coeffs(num_loc_dofs);
418  // Fetch global numbers of local shape functions
419  nonstd::span<const lf::assemble::gdof_idx_t> ldof_gidx(
420  dofh.GlobalDofIndices(*cell));
421  // Copy degrees of freedom into temporory vector
422  for (int j = 0; j < num_loc_dofs; ++j) {
423  loc_coeffs[j] = uh[ldof_gidx[j]];
424  }
425  // Sum local contribution
426  sum += loc_comp(*cell, loc_coeffs);
427  } // end if (isACtive())
428  }
429  return sum;
430 }
