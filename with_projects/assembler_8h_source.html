89 void AssembleMatrixLocally(dim_t codim, const DofHandler &dof_handler_trial,
90  const DofHandler &dof_handler_test,
91  ENTITY_MATRIX_PROVIDER &entity_matrix_provider,
92  TMPMATRIX &matrix) {
93  // Pointer to underlying mesh
94  auto mesh = dof_handler_trial.Mesh();
95 
96  LF_ASSERT_MSG(mesh == dof_handler_test.Mesh(),
97  "Trial and test space must be defined on the same mesh");
98 
99  // Central assembly loop over entities of co-dimension specified by
100  // the function argument codim
101  for (const lf::mesh::Entity *entity : mesh->Entities(codim)) {
102  // Some entities may be skipped
103  if (entity_matrix_provider.isActive(*entity)) {
104  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_entity,
105  std::cout << "ASM: " << *entity << '('
106  << mesh->Index(*entity) << ')' << std::endl);
107  // Size, aka number of rows and columns, of element matrix
108  const size_type nrows_loc = dof_handler_test.NumLocalDofs(*entity);
109  const size_type ncols_loc = dof_handler_trial.NumLocalDofs(*entity);
110  // row indices of for contributions of cells
111  nonstd::span<const gdof_idx_t> row_idx(
112  dof_handler_test.GlobalDofIndices(*entity));
113  // Column indices of for contributions of cells
114  nonstd::span<const gdof_idx_t> col_idx(
115  dof_handler_trial.GlobalDofIndices(*entity));
116  // Request local matrix from entity_matrix_provider object. In the case
117  // codim = 0, when `entity` is a cell, this is the element matrix
118  const auto elem_mat{entity_matrix_provider.Eval(*entity)};
119  LF_ASSERT_MSG(elem_mat.rows() >= nrows_loc,
120  "nrows mismatch " << elem_mat.rows() << " <-> " << nrows_loc
121  << ", entity " << mesh->Index(*entity));
122  LF_ASSERT_MSG(elem_mat.cols() >= ncols_loc,
123  "ncols mismatch " << elem_mat.cols() << " <-> " << nrows_loc
124  << ", entity " << mesh->Index(*entity));
125  // clang-format off
126  SWITCHEDSTATEMENT(
127  ass_mat_dbg_ctrl, amd_gdof,
128  std::cout << "ASM: row_idx = ";
129  for (auto gdof_idx: row_idx) {
130  std::cout << gdof_idx << ' ';
131  }
132  std::cout << std::endl << "ASM: col_idx = ";
133  for (auto gdof_idx : col_idx) {
134  std::cout << gdof_idx << ' ';
135  }
136  std::cout << std::endl);
137  // clang-format on
138  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_lmdim,
139  std::cout << "ASM: " << nrows_loc << " x " << ncols_loc
140  << " element matrix" << std::endl);
141  SWITCHEDSTATEMENT(
142  ass_mat_dbg_ctrl, amd_locmat, for (int i = 0; i < nrows_loc; i++) {
143  std::cout << "[ ";
144  for (int j = 0; j < ncols_loc; j++) {
145  std::cout << elem_mat(i, j) << ' ';
146  }
147  std::cout << "]" << std::endl;
148  });
149  // Assembly double loop
150  for (int i = 0; i < nrows_loc; i++) {
151  for (int j = 0; j < ncols_loc; j++) {
152  // Add the element at position (i,j) of the local matrix
153  // to the entry at (row_idx[i], col_idx[j]) of the global matrix
154  matrix.AddToEntry(row_idx[i], col_idx[j], elem_mat(i, j));
155  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_lass,
156  std::cout << "(" << row_idx[i] << ',' << col_idx[j]
157  << ")+= " << elem_mat(i, j) << ", ";);
158  }
159  } // end assembly local double loop
160  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_lass, std::cout << std::endl;);
161  } // end if(isActive() )
162  } // end main assembly loop
163 } // end AssembleMatrixLocally

247 template <typename VECTOR, class ENTITY_VECTOR_PROVIDER>
248 void AssembleVectorLocally(dim_t codim, const DofHandler &dof_handler,
249  ENTITY_VECTOR_PROVIDER &entity_vector_provider,
250  VECTOR &resultvector) {
251  // Pointer to underlying mesh
252  auto mesh = dof_handler.Mesh();
253 
254  // Central assembly loop over entities of the co-dimension specified via
255  // the template argument CODIM
256  for (const lf::mesh::Entity *entity : mesh->Entities(codim)) {
257  // Some cells may be skipped
258  if (entity_vector_provider.isActive(*entity)) {
259  // Length of element vector
260  const size_type veclen = dof_handler.NumLocalDofs(*entity);
261  // global dof indices for contribution of the entity
262  nonstd::span<const gdof_idx_t> dof_idx(
263  dof_handler.GlobalDofIndices(*entity));
264  // Request local vector from entity_vector_provider object. In the case
265  // CODIM = 0, when `entity` is a cell, this is the element vector
266  const auto elem_vec{entity_vector_provider.Eval(*entity)};
267  LF_ASSERT_MSG(elem_vec.size() >= veclen,
268  "length mismatch " << elem_vec.size() << " <-> " << veclen
269  << ", entity " << mesh->Index(*entity));
270  // Assembly (single) loop
271  for (int i = 0; i < veclen; i++) {
272  resultvector[dof_idx[i]] += elem_vec[i];
273  } // end assembly localloop
274  } // end if(isActive() )
275  } // end main assembly loop
276 } // end AssembleVectorLocally
