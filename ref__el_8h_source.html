404  [[nodiscard]] constexpr sub_idx_t SubSubEntity2SubEntity(
405  dim_t sub_codim, sub_idx_t sub_index, dim_t sub_rel_codim,
406  sub_idx_t sub_rel_index) const {
407  LF_ASSERT_MSG_CONSTEXPR(sub_codim >= 0, "sub_codim negative");
408  LF_ASSERT_MSG_CONSTEXPR(sub_codim <= Dimension(), "sub_codim > Dimension");
409  LF_ASSERT_MSG_CONSTEXPR(sub_index >= 0, "sub_index negative");
410  LF_ASSERT_MSG_CONSTEXPR(sub_index <= NumSubEntities(sub_codim),
411  "sub_index >= NumSubEntities");
412  LF_ASSERT_MSG_CONSTEXPR(sub_rel_codim >= 0, "sub_rel_codim negative.");
413  LF_ASSERT_MSG_CONSTEXPR(sub_rel_codim <= Dimension() - sub_codim,
414  "subSubCodim out of bounds.");
415  LF_ASSERT_MSG_CONSTEXPR(sub_rel_index >= 0, "sub_rel_index negative.");
416  LF_ASSERT_MSG_CONSTEXPR(
417  sub_rel_index <
418  SubType(sub_codim, sub_index).NumSubEntities(sub_rel_codim),
419  "sub_sub_index out of bounds.");
420 
421  if (type_ == RefElType::kPoint) {
422  return 0;
423  }
424  if (sub_codim == 0) {
425  return sub_rel_index;
426  }
427  if (sub_codim == Dimension()) {
428  return sub_index;
429  }
430 
431  // from here on, it must be a segment
432  switch (type_) {
433  case RefElType::kTria:
434  return sub_sub_entity_index_tria_[sub_index][sub_rel_index];
435  case RefElType::kQuad:
436  return sub_sub_entity_index_quad_[sub_index][sub_rel_index];
437  default:
438  LF_ASSERT_MSG_CONSTEXPR(false, "This code should never be reached.");
439  }
440 
441  return 0; // Prevent warnings from compiler...
442  }
